document.getElementById("startScanner").addEventListener("click", function () { qrCodeReader.start( { facingMode: "environment" }, // Use back camera by default { fps: 10, // Scanning speed qrbox: { width: 250, height: 250 } // QR Code scanning area }, (decodedText, decodedResult) => { alert(`Scanned content: ${decodedText}`); qrCodeReader.stop(); // Automatically stop the scanner after successful scan document.getElementById("reader").style.display = "none"; document.getElementById("startScanner").style.display = "block"; document.getElementById("stopScanner").style.display = "none"; }, (errorMessage) => { console.log(`QR Code scan error: ${errorMessage}`); } ).then(() => { document.getElementById("reader").style.display = "block"; document.getElementById("startScanner").style.display = "none"; document.getElementById("stopScanner").style.display = "block"; }).catch(err => { console.error(`Failed to start scanner: ${err}`); alert("Unable to access the camera. Please check permissions."); }); }); // Stop the QR Code Scanner document.getElementById("stopScanner").addEventListener("click", function () { qrCodeReader.stop().then(() => { document.getElementById("reader").style.display = "none"; document.getElementById("startScanner").style.display = "block"; document.getElementById("stopScanner").style.display = "none"; }).catch(err => { console.error("Error stopping scanner: ", err); }); });
